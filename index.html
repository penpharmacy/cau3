แบบประเมินผู้ป่วยรายเคส

1. ประเมินความจำเป็นในการใส่สายสวนปัสสาวะ ( เวรเช้าทำ )

2. ใช้ hand hygiene เมื่อสัมผัสสายสวนปัสสาวะ

3. ให้ระบบปัสสาวะเป็น close system ตลอดเวลา

4. ให้สายสวนปัสสาวะอยู่ในตำแหน่งที่เหมาะสมเพื่อให้ปัสสาวะไหลได้สะดวก

(ตลอดเวลา ไม่มีการอุดตัน สายไม่งอพับ)

5. จัดวางตำแหน่งของถุงเก็บปัสสาวะแขวนไว้ต่ำกว่าระดับกระเพาะปัสสาวะ โดยไม่สัมผัสพื้น

6. ไม่เปลี่ยนสายสวนปัสสาวะเป็น Routine

(เปลี่ยนเมื่อมีการรั่วซึมหรืออุดตัน โดยต้องทำการเปลี่ยนทั้งชุดสวนปัสสาวะ)

7. ทำความสะอาดบริเวณ urethral meatus เช้า – เย็นและหลังถ่ายอุจจาระทุกครั้งด้วย NSS

8. การเทปัสสาวะทำความสะอาดบริเวณ Urinary drainage tap ด้วย 70% alcohol

(ใช้ภาชนะที่สะอาด หลีกเลี่ยงอย่าให้บริเวณ Urinary drainage tap สัมผัสภาชนะที่รองรับ)

const SHEET_NAME = "CAUTi";

function doGet() {
  return HtmlService.createHtmlOutputFromFile('Index')
      .setTitle('CAUTI Care Bundle Checklist - พุนพิน')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL)
      .addMetaTag('viewport', 'width=device-width, initial-scale=1');

function getTargetSheet() {
  var ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  var sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
  }
  return sheet;
}

// ดึงข้อมูลทั้งหมดมาวิเคราะห์บน Dashboard
function getData() {
  try {
    var sheet = getTargetSheet();
    if (sheet.getLastRow() <= 1) return { headers: [], rows: [] };
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var rows = data.slice(1);
    
    var formattedRows = rows.map(function(row) {
      return row.map(function(cell) {
        if (cell instanceof Date) {
          return cell.toISOString().split('T')[0];
        }
        return cell;
      });
    });
    return { headers: headers, rows: formattedRows };
  } catch(e) {
    return { headers: [], rows: [] };
  }
}
